59 int positionToInt(const std::string& pos) {
60 if (pos.length() != 2) {
61 throw std::invalid_argument("Invalid position length");
62 }
63
64 char file = pos[0];
65 char rank = pos[1];
66
67 if (file < 'a' || file > 'h' || rank < '1' || rank > '8') {
68 throw std::invalid_argument("Invalid position format");
69 }
70
71 int fileIndex = file - 'a'; // 'a' -> 0, 'b' -> 1, ..., 'h' -> 7
72 int rankIndex = rank - '1'; // '1' -> 0, '2' -> 1, ..., '8' -> 7
73
74 return rankIndex * 8 + fileIndex; // 0-63
75 }

81 Move(const std::string& notation)
82 : from(-1), to(-1), promotion('\0') {
83 if (notation.length() == 4) { // regular move
84 from = positionToInt(notation.substr(0, 2));

86 } else if (notation.length() == 5) { // pawn promotion
87 from = positionToInt(notation.substr(0, 2));
